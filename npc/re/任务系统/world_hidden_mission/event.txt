// ===== 【世界隐藏任务】【神之力】：事件触发器 ================================ 


// 监听世界隐藏任务事件
-	script	#WORLD_HIDDEN_MISSION	-1,{

	// 当玩家死亡时，重置所有任务进度：
	OnPCDieEvent:
		callfunc("f_un_all_god_equipments");	// 卸下所有修炼神装
		callfunc("f_reset_city_checkpoints");	// 重置所有城市的游历标记
		end;

	// 当玩家状态统计数据变更时（如过图、换装备、升级等）：
	OnPCStatCalcEvent:
		callfunc("F_CHECK_SHOES");				// 触发并检查【神之力】鞋子的修炼进度
		end;
}
